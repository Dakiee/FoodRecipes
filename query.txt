CREATE TABLE ingredients(
    ingredientsId int NOT NULL AUTO_INCREMENT,
    name varchar(100) NOT NULL,
    type varchar(100),
    origin varchar(100),
    PRIMARY KEY(ingredientsId)
);

CREATE TABLE Recipes( 
    RecipeId int NOT NULL AUTO_INCREMENT, 
    RecipeName varchar(100) NOT NULL, 
    CookTime varchar(100), 
    ingredientsId int,
    PRIMARY KEY(RecipeId), 
    FOREIGN KEY (ingredientsId) 
    REFERENCES ingredients(ingredientsId), 
    desription varchar(100) 
);

CREATE TABLE favorites( 
    favId int NOT NULL AUTO_INCREMENT, 
    dates date, 
    recipeId int, 
    PRIMARY KEY(favId), 
    FOREIGN KEY (recipeId) REFERENCES recipes(recipeId) 
);

CREATE TABLE users( 
    UserID int NOT NULL AUTO_INCREMENT, 
    username varchar(100) NOT NULL, 
    PASSWORD varchar(100) NOT NULL,
    favId int,
    PRIMARY KEY(UserID), 
    FOREIGN KEY (favId) REFERENCES favorites(favId) 
);
